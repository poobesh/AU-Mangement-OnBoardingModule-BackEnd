CREATE TABLE `accolite`.`employee_constant` ( `id` INT NOT NULL , `first_name` VARCHAR(15) NOT NULL , `last_name` VARCHAR(15) NOT NULL , `email` VARCHAR(50) NOT NULL , `dob` DATE NOT NULL , `blood_type` VARCHAR(5) NOT NULL , `gender` VARCHAR(6) NOT NULL , `date_of_joining` DATE NOT NULL , `permanent_address` TEXT NOT NULL , `pincode` INT NOT NULL , `pan_number` VARCHAR(10) NOT NULL , `version` INT NOT NULL DEFAULT '0' , PRIMARY KEY (`id`)) ENGINE = InnoDB COMMENT = 'Employee Details that are not supposed to be changed';
CREATE TABLE `accolite`.`demand` ( `demand_id` INT NOT NULL , `hiring_manager_id` INT NOT NULL , `company_name` VARCHAR(30) NOT NULL , `location` VARCHAR(15) NOT NULL , `skillset` VARCHAR(30) NOT NULL , `joining_date` DATE NOT NULL , `status` VARCHAR(6) NOT NULL , `posted_date` DATE NOT NULL , PRIMARY KEY (`demand_id`)) ENGINE = InnoDB COMMENT = 'Demand Table Gives the demands that are active and closed';
CREATE TABLE `accolite`.`employee_editable` ( `email` VARCHAR(50) NOT NULL , `version` INT NOT NULL , `skill_id` INT NOT NULL , `experience` INT NOT NULL , `phone_number` INT NOT NULL , `current_address` TEXT NOT NULL , `pincode` INT NOT NULL , `BGC` BOOLEAN NOT NULL , `designation` VARCHAR(20) NOT NULL , `bank_ac_no` BIGINT NOT NULL , `demand_id` INT NULL DEFAULT NULL , PRIMARY KEY (`email`, `version`)) ENGINE = InnoDB COMMENT = 'Employee Editable table contains fields that are editable';
CREATE TABLE `accolite`.`address` ( `pincode` INT NOT NULL ,  `district` VARCHAR(20) NOT NULL ,  `state` VARCHAR(20) NOT NULL ,  `country` VARCHAR(20) NOT NULL ,    PRIMARY KEY  (`pincode`)) ENGINE = InnoDB COMMENT = 'Address table contains the state and district details';
CREATE TABLE `accolite`.`bank_details` ( `ac_no` BIGINT NOT NULL , `ifsc_code` VARCHAR(10) NOT NULL , `name` VARCHAR(30) NOT NULL , `branch` VARCHAR(30) NOT NULL , PRIMARY KEY (`ac_no`)) ENGINE = InnoDB COMMENT = 'Gives Bank Details';
CREATE TABLE `accolite`.`skill` ( `skill_id` INT NOT NULL , `skill_1` VARCHAR(30) NOT NULL , `skill_2` VARCHAR(30) NOT NULL , `skill_3` VARCHAR(30) NOT NULL , PRIMARY KEY (`skill_id`)) ENGINE = InnoDB;
CREATE TABLE `accolite`.`hiring_manager` ( `id` INT NOT NULL , `name` VARCHAR(30) NOT NULL , `phone_no` INT NOT NULL , `email` VARCHAR(50) NOT NULL , `team_name` VARCHAR(30) NOT NULL , `department_name` VARCHAR(30) NOT NULL , `required_employee_count` INT NOT NULL , `duration` INT NOT NULL , `required_experience` INT NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB COMMENT = 'Gives Details About Hiring Manager';
